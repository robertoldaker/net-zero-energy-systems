server {
    listen 80;
    server_name lv-app.net-zero-energy-systems.org lv-app-tmp.net-zero-energy-systems.org;
    location / {
        if ( -f /home/rob/websites/SmartEnergyToolsGui/app_offline.html ) {
                return 503;
        }
        proxy_pass http://localhost:5021;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection keep-alive;
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
		proxy_read_timeout 1800;
    }

    error_page 503 /app_offline.html;
    location = /app_offline.html {
        root /home/rob/websites/SmarteEnergyToolsGui/;
    }
