server {
    listen: 80
    server_name lv-data.net-zero-energy-systems.org;
    location / {
        if ( -f /home/rob/websites/SmartEnergyToolsDataApi/app_offline.html ) {
                return 503;
        }
        proxy_pass http://localhost:5020;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection keep-alive;
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
		proxy_read_timeout 1800;
		client_max_body_size 200M;
    }

    # special config for signal-r hub
    location /NotificationHub {
        proxy_pass http://localhost:5020;

        # Configuration for WebSockets
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection upgrade;
        proxy_cache off;

        # Configuration for ServerSentEvents
        proxy_buffering off;

        # Configuration for LongPolling or if your KeepAliveInterval is longer than 60 seconds
        proxy_read_timeout 100s;

        proxy_set_header Host $host;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
	
    error_page 503 /app_offline.html;
    location = /app_offline.html {
        root /home/rob/websites/SmartEnergyToolsDataApi/;
    }
}
